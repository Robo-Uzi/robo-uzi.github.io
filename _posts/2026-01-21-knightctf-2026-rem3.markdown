#!/usr/bin/env python3

from typing import List

FLAG_LEN = 0x1D  # 29 bytes

CONST_A = 0x2F910ED35CA71942
CONST_B = 0x6A124DE908B17733

TARGET = bytes.fromhex(
    "dc 6b bb 4d fd 25 e4 7e"
    "c3 26"
    "f5 72 ab 96 fc 8d 55 10 93 c1"
    "fd 81 46 5b 7e 33 83 8f 2f"
)

PRINTABLE = range(0x20, 0x7F)


def rol8(value: int, shift: int) -> int:
    shift &= 7
    return ((value << shift) | (value >> (8 - shift))) & 0xFF


def ror8(value: int, shift: int) -> int:
    shift &= 7
    return ((value >> shift) | (value << (8 - shift))) & 0xFF


def encrypt_in_place(buf: bytearray) -> None:

    rolling_xor = 0               # bVar9
    rolling_add = 0               # cVar8
    signed_accumulator = -0x3D    # iVar7 (needs to stay signed int)

    for index in range(FLAG_LEN):
        shift = ((index & 7) << 3)

        # mix input byte
        const_byte = (CONST_A >> shift) & 0xFF
        mixed = (const_byte + rolling_add) & 0xFF
        mixed ^= buf[index]

        rolling_add = (rolling_add + 0x1D) & 0xFF

        # dynamic rotation
        rot_amount = (CONST_B >> ((index * 8 + 0x10) & 0x38)) & 7

        xor_mask = ((CONST_B >> shift) & 0xFF) ^ rolling_xor
        rolling_xor = (rolling_xor + 0x11) & 0xFF
